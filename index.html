Dumb button

<a class="btn" id="singleButton">Dumb button</a> 
d3.select('#slide3 #singleButton')
 .text('POWER button!')
 .classed('btn-primary', true)
 .style('font-weight', 'bold')
 .on('click', function() {
  d3.select(this)
   .text('Dumb button')
   .classed('blueish', false)
   .style('font-weight:normal');
 });
Execute
